-- ตารางข้อมูลคนไข้
CREATE TABLE patient (
    id INT AUTO_INCREMENT PRIMARY KEY,         			-- id PRIMARY Key
    HN VARCHAR(20) NOT NULL UNIQUE,            			-- รหัสคนไข้
    name VARCHAR(100),                         			-- ชื่อ
    national_id VARCHAR(13),                   			-- รหัสประจำตัวประชาชน
    age INT,                                   			-- อายุ
    gender VARCHAR(10),                        			-- เพศ
    address TEXT,                              			-- ที่อยู่
    height DECIMAL(5,2),                       			-- ส่วนสูง
    weight DECIMAL(5,2),                       			-- น้ำหนัก
    bmi DECIMAL(5,2),                          			-- BMI
    blood_pressure VARCHAR(50),                			-- ความดันโลหิต/ชีพจร
    chronic_diseases TEXT,                     			-- โรคประจำตัว
    reason_for_visit TEXT,                     			-- สาเหตุที่มาคลินิค.
    treatment_history TEXT                     			-- ประวัติการรักษา
);

-- ตารางประวัติการรักษา (เก็บข้อมูลเก่าและอัปเดตข้อมูลได้)
CREATE TABLE treatment_history (
    id INT AUTO_INCREMENT PRIMARY KEY,         			-- id PRIMARY Key
    HN VARCHAR(20),                            			-- รหัสคนไข้
    name VARCHAR(100),                         			-- ชื่อ
    record TEXT,                               			-- ประวัติการรักษา
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 		-- เวลาที่บันทึกข้อมูล
    FOREIGN KEY (HN) REFERENCES patient(HN)  			-- เชื่อมโยงกับตารางข้อมูลคนไข้โดยใช้ HN
);

-- ตารางประวัติการแพ้ยา (เก็บข้อมูลเก่าและอัปเดตข้อมูลได้)
CREATE TABLE medication_history (
    id INT AUTO_INCREMENT PRIMARY KEY,         			-- id PRIMARY Key
    HN VARCHAR(20),                            			-- รหัสคนไข้
    name VARCHAR(100),                         			-- ชื่อ
    record TEXT,                               			-- ประวัติการแพ้ยา
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 		-- เวลาที่บันทึกข้อมูล
    FOREIGN KEY (HN) REFERENCES patient(HN)  			-- เชื่อมโยงกับตารางข้อมูลคนไข้โดยใช้ HN
);

-- ตารางข้อมูลบุคลากร
CREATE TABLE users (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    role VARCHAR(50) NOT NULL
);
